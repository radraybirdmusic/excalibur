<script>
    const params = new URLSearchParams(window.location.search);
    const myNodeID = params.get('node') || "01";
    
    async function updateHUD() {
        try {
            const res = await fetch(`./manifest.json?t=${new Date().getTime()}`);
            const data = await res.json();
            const round = data.current_live_round;
            const nodeName = data.assignments[myNodeID][0];
            const schedule = data.round_schedule[round][nodeName];

            // 1. Identity & Round
            document.getElementById('round-num').innerText = round;
            document.getElementById('player-role').innerText = nodeName.replace('NODE_', '');
            document.getElementById('player-bio').innerText = data.library[nodeName].bio;
            document.getElementById('system-status').innerText = `${data.system_status} // SYNC_OK`;

            // 2. Motives (White for contrast)
            let mHTML = "";
            schedule.motives.forEach(m => { mHTML += `<li>${m}</li>`; });
            document.getElementById('motive-list').innerHTML = mHTML;

            // 3. Alliances & Inventory (Using the metadata we will add later)
            // For now, let's pull 'clues' into inventory
            let iHTML = "";
            if(schedule.clues) {
                schedule.clues.forEach(c => { iHTML += `<span class="item-tag">${c}</span>`; });
            }
            document.getElementById('inventory-list').innerHTML = iHTML || "EMPTY";
            
            // Heat Status (Example check)
            document.getElementById('heat-status').innerText = data.system_status.includes("STABLE") ? "LOW" : "ELEVATED";

            // 4. Ledger Logic
            const ledger = document.getElementById('quest-ledger');
            if (data.ledger_visibility === "VISIBLE") {
                ledger.style.display = "block";
                let qHTML = "<ul>";
                for (let n in data.round_schedule[round]) {
                    qHTML += `<li><strong>${n.replace('NODE_', '')}:</strong> ${data.round_schedule[round][n].motives[0]}</li>`;
                }
                document.getElementById('quest-list').innerHTML = qHTML + "</ul>";
            } else { ledger.style.display = "none"; }

        } catch (e) { document.getElementById('system-status').innerText = "CONNECTION LOST..."; }
    }
    updateHUD();
    setInterval(updateHUD, 10000);
</script>
